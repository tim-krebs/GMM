95int GMM::Expectation_Maximation(const std::vector<std::vector<double> > melCepData, size_t frameCount)
96{
97 int step;
98 int iteration = 0;
99 double newProb;
100 double recentProb = 0.0;
101
102 std::vector<double> sumProb(m_MixDim);
103 std::vector<double> mixedProb(frameCount);
104 std::vector<std::vector<double>> normProb(frameCount, std::vector<double>(m_MixDim));
105 std::vector<std::vector<double>> tempProb(m_MfccDim, std::vector<double>(m_MixDim));
106 std::vector<std::vector<double>> squareCep(frameCount, std::vector<double>(m_MfccDim));
107
108 //*** Initialization
109 for(int i = 0; i < m_MixDim; i++)
110 {
111 m_Model.weight[i] = 1.0 / m_MixDim;
112 }
113
114
115 step = (int)floor(frameCount / m_MixDim);
116
117 for(int j = 0; j < m_MixDim; j++)
118 {
119 for(int i = 0; i < m_MfccDim; i++)
120 {
121 m_Model.mean[i][j] = melCepData[step * (j + 1) -1][i];
122 }
123 }
124
125 for(int i = 0; i < m_MixDim; i++)
126 {
127 for(int j = 0; j < m_MfccDim; j++)
128 {
129 m_Model.covariance[i][j] = 1.0;
130 }
131 }
132
133 for(size_t i = 0; i < frameCount; i++)
134 {
135 for(int j = 0; j < m_MfccDim; j++)
136 {
137 squareCep[i][j] = melCepData[i][j] * melCepData[i][j];
138 }
139 }
140
141
142 // Iterative processing
143 // EM-Algorithm
144 while(true)
145 {
146 completeModel(m_Model);
147 newProb = Likelihood(melCepData, frameCount, m_Model, normProb, mixedProb);
148
149 // E process, a probability matrix, n*m_MixDim
150 for(size_t i = 0; i < frameCount; i++)
151 {
152 for(int j = 0; j < m_MixDim; j++)
153 {
154 normProb[i][j] *= m_Model.weight[j] / mixedProb[i];
155 }
156 }
157
158 // M process
159 for(int i = 0; i < m_MixDim; i++)
160 {
161 sumProb[i] = 0.0;
162 for(size_t j = 0; j < frameCount; j++)
163 {
164 sumProb[i] += normProb[j][i];
165 }
166 }
167
168 // renew mixture coefficients
169 for(int i = 0; i < m_MixDim; i++)
170 {
171 m_Model.weight[i] = sumProb[i] / frameCount;
172 }
173
174 // renew mean
175 for(int i = 0; i < m_MfccDim; i++)
176 {
177 for(int j = 0; j < m_MixDim; j++)
178 {
179 tempProb[i][j] = 0.0;
180 for(size_t k = 0; k < frameCount; k++)
181 tempProb[i][j] += melCepData[k][i] * normProb[k][j];
182
183 m_Model.mean[i][j] = tempProb[i][j] / sumProb[j];
184 }
185 }
186
187 // renew covariance
188 for(int i = 0; i < m_MixDim; i++)
189 {
190 for(int j = 0; j < m_MfccDim; j++)
191 {
192 tempProb[j][i] = 0.0;
193 for(size_t k = 0; k < frameCount; k++)
194 {
195 tempProb[j][i] += squareCep[k][j] * normProb[k][i];
196 }
197 tempProb[j][i] = tempProb[j][i] / sumProb[i];
198 }
199 }
200
201 // set covariance
202 for(int i = 0; i < m_MixDim; i++)
203 {
204 for(int j = 0; j < m_MfccDim; j++)
205 {
206 m_Model.covariance[i][j] = tempProb[j][i] - m_Model.mean[j][i] * m_Model.mean[j][i];
207 if(m_Model.covariance[i][j] <= m_MinCov)
208 {
209 m_Model.covariance[i][j] = m_MinCov;
210 }
211 }
212 }
213 // prepare for next iteration
214 recentProb = newProb;
215 iteration++;
216 if(iteration > 19) break;
217 }
218
219 tempProb.clear();
220 normProb.clear();
221 sumProb.clear();
222 mixedProb.clear();
223
224 return iteration;
225}

234std::string GMM::Classify(const std::vector<std::vector<double> >& melCepData, size_t frameCount)
235{

252 normProb[i].resize(m_MixDim);

463double GMM::Likelihood(const std::vector<std::vector<double> > &melCepData, size_t frameCount, Model model, std::vector<std::vector<double> > &normProb, std::vector<double> &mixedProb)
464{
465 double prob = 0.0;
466 std::vector<double> maxMatrix(frameCount);
467 std::vector<std::vector<double> > expMatrix(frameCount, std::vector<double>(m_MixDim, 0));
468
469 // Calculate the Expectation Matrix
470 for(size_t i = 0; i < frameCount; i++ )
471 {
472 for(int j = 0; j < m_MixDim; j++)
473 {
474 for(int k = 0; k < m_MfccDim; k++)
475 {
476 expMatrix[i][j] += (pow((melCepData[i][k] - model.mean[k][j]),2 )) * model.invert_covariance[j][k];
477 }
478 }
479 }
480
481 for(size_t i = 0; i < frameCount; i++)
482 {
483 auto it = max_element(expMatrix[i].begin(), expMatrix[i].end());
484 maxMatrix[i] = *it;
485 }
486
487 // calculate Probability for each frame
488 for(size_t i = 0; i < frameCount; i++)
489 {
490 mixedProb[i] = 0.0;
491 for(int j = 0; j < m_MixDim; j++)
492 {
493 expMatrix[i][j] = exp(expMatrix[i][j] - maxMatrix[i]);
494 normProb[i][j] = expMatrix[i][j] * model.ExpCoeff[j];
495 mixedProb[i] = mixedProb[i] + normProb[i][j] * model.weight[j];
496 }
497 prob += log(mixedProb[i]) + maxMatrix[i];
498 }
499
500 expMatrix.clear();
501 maxMatrix.clear();
502
503 return prob;
504}

557void GMM::completeModel(Model& model)
558{
559 double x = pow(PI2, (-m_MfccDim / 2));
560
561 for(int i = 0; i < m_MixDim; i++)
562 {
563 model.ExpCoeff[i] = 1.0;
564
565 for(int j = 0; j < m_MfccDim; j++)
566 {
567 model.ExpCoeff[i] *= 1.0 / model.covariance[i][j];
568 }
569
570 model.ExpCoeff[i] = x * sqrt(model.ExpCoeff[i]);
571 }
572
573 for(int i = 0; i < m_MixDim; i++)
574 {
575 for(int j = 0; j < m_MfccDim; j++)
576 {
577 model.invert_covariance[i][j] = (-0.5) / model.covariance[i][j];
578 }
