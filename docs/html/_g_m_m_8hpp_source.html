88int GMM::Expectation_Maximation(const std::vector<std::vector<double> > melCepData, size_t frameCount)
89{
90 int step;
91 int iteration = 0;
92 double newProb;
93 double recentProb = 0.0;
94
95 std::vector<double> sumProb(m_MixDim);
96 std::vector<double> mixedProb(frameCount);
97 std::vector<std::vector<double>> normProb(frameCount, std::vector<double>(m_MixDim));
98 std::vector<std::vector<double>> tempProb(m_MfccDim, std::vector<double>(m_MixDim));
99 std::vector<std::vector<double>> squareCep(frameCount, std::vector<double>(m_MfccDim));
100
101 //*** Initialization
102 for(int i = 0; i < m_MixDim; i++)
103 {
104 m_Model.weight[i]=1.0 / m_MixDim;
105 }
106
107 step= (int)floor(frameCount / m_MixDim);
108
109 for(int j = 0; j < m_MixDim; j++)
110 {
111 for(int i = 0; i < m_MfccDim; i++)
112 {
113 m_Model.mean[i][j] = melCepData[step*(j + 1) -1][i];
114 }
115 }
116
117 for(int i = 0; i < m_MixDim; i++)
118 {
119 for(int j = 0; j < m_MfccDim; j++)
120 {
121 m_Model.covariance[i][j] = 1.0;
122 }
123 }
124
125
126 for(size_t i = 0; i < frameCount; i++)
127 {
128 for(int j = 0; j < m_MfccDim; j++)
129 {
130 squareCep[i][j] = melCepData[i][j] * melCepData[i][j];
131 }
132 }
133
134
135 //*** Iterative processing
136 // EM-Algorithm
137 while(true)
138 {
139 completeModel(m_Model);
140 newProb = Probability(melCepData, frameCount, m_Model, normProb, mixedProb);
141
142 //if( ((newProb - recentProb) < (m_Threshold * frameCount)) && (iteration > 0) )
143 //{
144 // break;
145 //}
146
147 //*** E process, a probability matrix, n*m_MixDim
148 for(size_t i = 0; i < frameCount; i++)
149 {
150 for(int j = 0; j < m_MixDim; j++)
151 {
152 normProb[i][j] *= m_Model.weight[j] / mixedProb[i];
153 }
154 }
155
156 //*** M process
157 for(int i = 0; i < m_MixDim; i++)
158 {
159 sumProb[i] = 0.0;
160 for(size_t j = 0; j < frameCount; j++)
161 {
162 sumProb[i] += normProb[j][i];
163 }
164 }
165
166 //*** renew mixture coefficients
167 for(int i = 0; i < m_MixDim; i++)
168 {
169 m_Model.weight[i] = sumProb[i] / frameCount;
170 }
171
172 //*** renew mean
173 for(int i = 0; i < m_MfccDim; i++)
174 {
175 for(int j = 0; j < m_MixDim; j++)
176 {
177 tempProb[i][j] = 0.0;
178 for(size_t k = 0; k < frameCount; k++)
179 tempProb[i][j] += melCepData[k][i] * normProb[k][j];
180
181 m_Model.mean[i][j] = tempProb[i][j] / sumProb[j];
182 }
183 }
184
185 //*** renew covariance
186 for(int i = 0; i < m_MixDim; i++)
187 {
188 for(int j = 0; j < m_MfccDim; j++)
189 {
190 tempProb[j][i] = 0.0;
191 for(size_t k = 0; k < frameCount; k++)
192 {
193 tempProb[j][i] += squareCep[k][j] * normProb[k][i];
194 }
195 tempProb[j][i] = tempProb[j][i] / sumProb[i];
196 }
197 }
198
199 for(int i = 0; i < m_MixDim; i++)
200 {
201 for(int j = 0; j < m_MfccDim; j++)
202 {
203 m_Model.covariance[i][j] = tempProb[j][i] - m_Model.mean[j][i] * m_Model.mean[j][i];
204 if(m_Model.covariance[i][j] <= m_MinCov)
205 {
206 m_Model.covariance[i][j] = m_MinCov;
207 }
208 }
209 }
210 //*** prepare for next iteration
211 recentProb = newProb;
212 iteration++;
213 if(iteration > 19) break;
214 }
215
216 tempProb.clear();
217 normProb.clear();
218 sumProb.clear();
219 mixedProb.clear();
220
221 return iteration;
222}

425double GMM::Probability(const std::vector<std::vector<double> > &melCepData, size_t frameCount, Model model, std::vector<std::vector<double> > &normProb, std::vector<double> &mixedProb)
426{
427 double prob = 0.0;
428 std::vector<double> maxMatrix(frameCount);
429 std::vector<std::vector<double> > expMatrix(frameCount, std::vector<double>(m_MixDim, 0));
430
431
432 for(size_t i = 0; i < frameCount; i++ )
433 {
434 for(int j = 0; j < m_MixDim; j++)
435 {
436 for(int k = 0; k < m_MfccDim; k++)
437 {
438 expMatrix[i][j] += (pow((melCepData[i][k] - model.mean[k][j]),2 )) * model.invert_covariance[j][k];
439 }
440 }
441 }
442
443 for(size_t i = 0; i < frameCount; i++)
444 {
445 auto it = max_element(expMatrix[i].begin(), expMatrix[i].end());
446 maxMatrix[i] = *it;
447 }
448
449 // calculate Probability for each frame
450 for(size_t i = 0; i < frameCount; i++)
451 {
452 mixedProb[i] = 0.0;
453 for(int j = 0; j < m_MixDim; j++)
454 {
455 expMatrix[i][j] = exp(expMatrix[i][j] - maxMatrix[i]);
456 normProb[i][j] = expMatrix[i][j] * model.ExpCoeff[j];
457 mixedProb[i] = mixedProb[i] + normProb[i][j] * model.weight[j];
458 }
459 prob += log(mixedProb[i]) + maxMatrix[i];
460 }
461
462 expMatrix.clear();
463 maxMatrix.clear();
464
465 return prob;
466}

505void GMM::completeModel(Model& model)
506{
507 double x = pow(PI2, (-m_MfccDim / 2));
508
509 for(int i = 0; i < m_MixDim; i++)
510 {
511 model.ExpCoeff[i] = 1.0;
512
513 for(int j = 0; j < m_MfccDim; j++)
514 {
515 model.ExpCoeff[i] *= 1.0 / model.covariance[i][j];
516 }
517
518 model.ExpCoeff[i] = x * sqrt(model.ExpCoeff[i]);
519 }

521 for(int i = 0; i < m_MixDim; i++)
522 {
523 for(int j = 0; j < m_MfccDim; j++)
524 {
525 model.invert_covariance[i][j] = (-0.5) / model.covariance[i][j];
526 }
